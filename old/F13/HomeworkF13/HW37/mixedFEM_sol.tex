\[
-[u'(x)v(x)]_0^1+a(u,v)=(f,v)
\]
from which we can conclude that
\[
\alpha v(0)+a(u,v)=(f,v)
\]
since $v(1)=0$ and $u'(0)=\alpha$. Therefore,
\[
a(u,v)=g(f,\alpha,v)\mbox{ for all }v\in\{w\in C^2[0,1]:w(1)=0\}
\]
where
\[
g(f,\alpha,v)=(f,v)-\alpha v(0).
